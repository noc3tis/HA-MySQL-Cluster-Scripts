# 1. Exporter en nodos MySQL (Ejemplo Nodo 1)
sudo nano /etc/.exporter.cnf # (Contiene credenciales del
usuario exporter)
sudo chmod 644 /etc/.exporter.cnf
# Inserción en /etc/default/prometheus-mysqld-exporter:
# ARGS="--config.my-cnf /etc/.exporter.cnf"
sudo systemctl restart prometheus-mysqld-exporter
# 2. Prometheus en Servidor de Monitoreo
sudo apt install prometheus -y
sudo nano /etc/prometheus/prometheus.yml
# (Se agregó el job_name 'mysql_cluster_equipoK' con los 3
targets IP:9104)
sudo chown prometheus:prometheus
/etc/prometheus/prometheus.yml
sudo systemctl restart prometheus
# 3. Grafana
# Instalación vía APT (repositorio de Grafana Labs)
sudo systemctl start grafana-server
# Se configuró el Data Source "Prometheus" apuntando a la IP
de Prometheus (Puerto 9090).
# Se importó el Dashboard ID 14057 para la visualización del
clúster.
# 4. Configuración de Alertas (Alerting en Grafana)
# Se creó una regla de alerta de alta prioridad para
monitorear la caída de nodos:
# - Query (PromQL): mysql_up{job="mysql_cluster_equipoK"}
# - Condición Matemática: Si el valor es igual a 0 ($B == 0).
# - Tolerancia (Pending period): 1 minuto (1m) de evaluación
para evitar falsos positivos por micro-cortes de red.
# - Acción: El sistema cambia a estado "Firing" alertando
sobre la indisponibilidad del nodo.
